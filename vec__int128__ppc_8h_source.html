2483 vec_clzq (vui128_t vra)
2484 {

2486

2488  /*
2489  * Use the Vector Count Leading Zeros Double Word instruction to get
2490  * the count for the left and right vector halves. If the left vector
2491  * doubleword of the input is nonzero then only the left count is
2492  * included and we need to mask off the right count.
2493  * Otherwise the left count is 64 and we need to add 64 to the right
2494  * count.
2495  * After masking we sum across the left and right counts to
2496  * get the final 128-bit vector count (0-128).
2497  */
2498  vui64_t vt1, vt2, vt3;
2499  vui64_t vzero = { 0, 0 };
2500  vui64_t v64 = { 64, 64 };
2501 
2502  vt1 = vec_clzd ((vui64_t) vra);
2503  vt2 = (vui64_t) vec_cmplt(vt1, v64);
2504  vt3 = (vui64_t) vec_sld ((vui8_t) vzero, (vui8_t) vt2, 8);
2505  result = vec_andc (vt1, vt3);
2506  result = (vui64_t) vec_sums ((vi32_t) result, (vi32_t) vzero);
2507 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
2508  result = (vui64_t) vec_sld ((vui8_t) result, (vui8_t) result, 4);
2509 #endif
2510 #else
2511  /* vector clz instructions were introduced in power8. For power7 and
2512  * earlier, use the pveclib vec_clzw implementation. For a quadword
2513  * clz, this requires pre-conditioning the input before computing the
2514  * the word clz and sum across. */
2515  vui32_t c0, clz;
2516  vui32_t r32, gt32, gt32sr32, gt64sr64;
2517

2519  gt32 = (vui32_t) vec_cmpgt ((vui32_t) vra, c0);
2520  gt32sr32 = vec_sld (c0, gt32, 12);
2521  gt64sr64 = vec_sld (c0, gt32, 8);
2522  gt32 = vec_sld (c0, gt32, 4);
2523 
2524  gt32sr32 = vec_or (gt32sr32, gt32);
2525  gt64sr64 = vec_or (gt64sr64, (vui32_t) vra);
2526  r32 = vec_or (gt32sr32, gt64sr64);
2527 
2528  clz = vec_clzw (r32);
2529  result = (vui64_t) vec_sums ((vi32_t) clz, (vi32_t) c0);
2530 #endif
2531
