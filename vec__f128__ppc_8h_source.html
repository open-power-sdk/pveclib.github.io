189 /* __float128 was added in GCC 6.0. But only with -mfloat128.
190  Later compilers typedef __float128 to __ieee128 and
191  long double to __ibm128. The intent was to allow the switch of
192  long double from __ibm128 to __ieee128 (someday).
193 
194  Clang does not define __FLOAT128__ or __float128 without both
195  -mcu=power9 and -mfloat128.
196  So far clang does not support/define the __ibm128 type. */
197 #ifdef __FLOAT128__
198 typedef __float128 __Float128;
199 typedef __float128 __binary128;
200 typedef __float128 __ieee128;
201 #ifndef __clang__
202 typedef __ibm128 __IBM128;
203 #else
204 /* Clang does not define __ibm128 over IBM long double.
205  So defined it here. */
206 typedef long double __IBM128;
207 #endif
208 #else
209 /* Before GCC 6.0 (or without -mfloat128) we need to fake it. */
212 typedef vui128_t vf128_t;
215 typedef vf128_t __Float128;
218 typedef vf128_t __binary128;
219 #ifndef __clang__
220 // Clang will not allow redefining __float128 even is it not enabled
223 typedef vf128_t __float128;
224 #endif
225 
227 typedef long double __IBM128;
228 #endif
