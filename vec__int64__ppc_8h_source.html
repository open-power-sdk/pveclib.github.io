1041 static inline
1042 vb64_t
1043 vec_cmpequd (vui64_t a, vui64_t b)
1044 {
1045  vb64_t result;
1046 #ifdef _ARCH_PWR8
1047 #if __GNUC__ >= 6
1048  result = vec_cmpeq(a, b);
1049 #else
1050  __asm__(
1051  "vcmpequd %0,%1,%2;\n"
1052  : "=v" (result)
1053  : "v" (a),
1054  "v" (b)
1055  : );
1056 #endif
1057 #else
1058  /*
1059  * Don't have vector compare equal unsigned doubleword until power8.
1060  * So we have to compare word and unless all_eq we need to do some
1061  * extra work, ie the words may have different truth values. So we
1062  * rotate each doubleword by 32-bits (here we use permute as we don't
1063  * have rotate doubleword either). Then vand the original word
1064  * compare and rotated value to get the final value.
1065  */
1066  vui8_t permute =
1067  { 0x04,0x05,0x6,0x7, 0x00,0x01,0x2,0x03, 0x0C,0x0D,0x0E,0x0F, 0x08,0x09,0x0A,0x0B};
1068  vui32_t r, rr;
1069  r = (vui32_t) vec_cmpeq ((vui32_t) a, (vui32_t) b);
1070  if (vec_any_ne ((vui32_t) a, (vui32_t) b))
1071  {
1072  rr = vec_perm (r, r, permute);
1073  r= vec_and (r, rr);
1074  }
1075  result = (vb64_t)r;
1076 #endif
1077  return (result);
1078 }

1212 static inline
1213 vb64_t
1214 vec_cmpgtud (vui64_t a, vui64_t b)
1215 {
1216  vb64_t result;
1217 #ifdef _ARCH_PWR8
1218 #if __GNUC__ >= 6
1219  result = vec_cmpgt(a, b);
1220 #else
1221  __asm__(
1222  "vcmpgtud %0,%1,%2;\n"
1223  : "=v" (result)
1224  : "v" (a),
1225  "v" (b)
1226  : );
1227 #endif
1228 #else
1229  /*
1230  * Don't have vector compare greater than unsigned doubleword until
1231  * power8. So we have to use compare word and logic to compute the
1232  * doubleword truth values.
1233  */
1234  __vector unsigned int r, x, y;
1235  __vector unsigned int c0, c1, c01;
1236  __vector unsigned int eq, gt, a32, b32;
1237 
1238  /* c10 = {0, -1, 0, -1} */
1239  c0 = vec_splat_u32 (0);
1240  c1 = vec_splat_u32 (-1);
1241  c01 = vec_mergeh (c0, c1);
1242 
1243  a32 = (__vector unsigned int)a;
1244  b32 = (__vector unsigned int)b;
1245 
1246  gt = (__vector unsigned int)vec_cmpgt (a32, b32);
1247  eq = (__vector unsigned int)vec_cmpeq (a32, b32);
1248  /* GTxw = GThw | (EQhw & GTlw) */
1249  x = vec_sld (gt, c0, 4);
1250  y = vec_and (eq, x);
1251  x = vec_or (gt, y);
1252  /* Duplicate result word to dword width. */
1253  y = vec_sld (c0, x, 12);
1254  r = vec_sel (x, y, c01);
1255  result = (vb64_t)r;
1256 #endif
1257  return (result);
1258 }
