552 static inline vui16_t
553 vec_clzh (vui16_t vra)
554 {
555  vui16_t r;
556 #ifdef _ARCH_PWR8
557 #if defined (vec_vclzh)
558  r = vec_vclzh (vra);
559 #elif defined (__clang__)
560  r = vec_cntlz (vra);
561 #else
562  __asm__(
563  "vclzh %0,%1;"
564  : "=v" (r)
565  : "v" (vra)
566  : );
567 #endif
568 #else
569 //#warning Implememention pre power8
570  vui16_t n, nt, y, x, s, m;
571  vui16_t z= { 0,0,0,0, 0,0,0,0};
572  vui16_t one = { 1,1,1,1, 1,1,1,1};
573 
574  /* n = 16 s = 8 */
575  s = vec_splat_u16(8);
576  n = vec_add (s, s);
577  x = vra;
578 
579  /* y=x>>8 if (y!=0) (n=n-8 x=y) */
580  y = vec_sr(x, s);
581  nt = vec_sub(n,s);
582  m = (vui16_t)vec_cmpgt(y, z);
583  s = vec_sr(s,one);
584  x = vec_sel (x , y, m);
585  n = vec_sel (n , nt, m);
586 
587  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
588  y = vec_sr(x, s);
589  nt = vec_sub(n,s);
590  m = (vui16_t)vec_cmpgt(y, z);
591  s = vec_sr(s,one);
592  x = vec_sel (x , y, m);
593  n = vec_sel (n , nt, m);
594 
595  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
596  y = vec_sr(x, s);
597  nt = vec_sub(n,s);
598  m = (vui16_t)vec_cmpgt(y, z);
599  s = vec_sr(s,one);
600  x = vec_sel (x , y, m);
601  n = vec_sel (n , nt, m);
602 
603  /* y=x>>1 if (y!=0) return (n=n-2) */
604  y = vec_sr(x, s);
605  nt = vec_sub(n,s);
606  nt = vec_sub(nt,s);
607  m = (vui16_t)vec_cmpgt(y, z);
608  n = vec_sel (n , nt, m);
609 
610  /* else return (x-n) */
611  nt = vec_sub (n, x);
612  r = vec_sel (nt , n, m);
613 #endif
614 
615  return (r);
616 }

833 static inline vui16_t
834 vec_popcnth (vui16_t vra)
835 {
836  vui16_t r;
837 #ifdef _ARCH_PWR8
838 #if defined (vec_vpopcnth)
839  r = vec_vpopcnth (vra);
840 #elif defined (__clang__)
841  r = vec_popcnt (vra);
842 #else
843  __asm__(
844  "vpopcnth %0,%1;"
845  : "=v" (r)
846  : "v" (vra)
847  : );
848 #endif
849 #else
850  //#warning Implememention pre power8
851  __vector unsigned short n, x1, x2, x, s;
852  __vector unsigned short ones = { 1,1,1,1, 1,1,1,1};
853  __vector unsigned short fives =
854  {0x5555,0x5555,0x5555,0x5555, 0x5555,0x5555,0x5555,0x5555};
855  __vector unsigned short threes =
856  {0x3333,0x3333,0x3333,0x3333, 0x3333,0x3333,0x3333,0x3333};
857  __vector unsigned short fs =
858  {0x0f0f,0x0f0f,0x0f0f,0x0f0f, 0x0f0f,0x0f0f,0x0f0f,0x0f0f};
859  /* n = 8 s = 4 */
860  s = ones;
861  x = vra;
862 
863  /* x = x - ((x >> 1) & 0x5555) */
864  x2 = vec_and (vec_sr (x, s), fives);
865  n = vec_sub (x, x2);
866  s = vec_add (s, s);
867 
868  /* x = (x & 0x3333) + ((x & 0xcccc) >> 2) */
869  x1 = vec_and (n, threes);
870  x2 = vec_andc (n, threes);
871  n = vec_add (x1, vec_sr (x2, s));
872  s = vec_add (s, s);
873 
874  /* x = (x + (x >> 4)) & 0x0f0f) */
875  x1 = vec_add (n, vec_sr (n, s));
876  n = vec_and (x1, fs);
877  s = vec_add (s, s);
878 
879  /* This avoids the extra load const. */
880  /* x = (x + (x << 8)) >> 8) */
881  x1 = vec_add (n, vec_sl (n, s));
882  r = vec_sr (x1, s);
883 #endif
884  return (r);
885 }

951 static inline vui16_t
952 vec_slhi (vui16_t vra, const unsigned int shb)
953 {
954  vui16_t lshift;
955  vui16_t result;
956 
957  if (shb < 16)
958  {
959  /* Load the shift const in a vector. The element shifts require
960  a shift amount for each element. For the immediate form the
961  shift constant is splatted to all elements of the
962  shift control. */
963  if (__builtin_constant_p(shb))
964  lshift = (vui16_t) vec_splat_s16(shb);
965  else
966  lshift = vec_splats ((unsigned short) shb);
967 
968  /* Vector Shift right bytes based on the lower 4-bits of
969  corresponding element of lshift. */
970  result = vec_vslh (vra, lshift);
971  }
972  else
973  { /* shifts greater then 15 bits return zeros. */
974  result = vec_xor ((vui16_t) vra, (vui16_t) vra);
975  }
976 
977  return (vui16_t) result;
978 }
