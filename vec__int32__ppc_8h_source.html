464 static inline vui32_t
465 vec_clzw (vui32_t vra)
466 {
467  vui32_t r;
468 #ifdef _ARCH_PWR8
469 #if defined (vec_vclzw)
470  r = vec_vclzw (vra);
471 #elif defined (__clang__)
472  r = vec_cntlz (vra);
473 #else
474  __asm__(
475  "vclzw %0,%1;"
476  : "=v" (r)
477  : "v" (vra)
478  : );
479 #endif
480 #else
481 //#warning Implememention pre POWER8
482  vui32_t n, nt, y, x, s, m;
483  vui32_t z= {0,0,0,0};
484  vui32_t one = {1,1,1,1};
485 
486  /* n = 32 s = 16 */
487  s = vec_splat_u32(8);
488  s = vec_add (s, s);
489  n = vec_add (s, s);
490 
491  x = vra;
492  /* y=x>>16 if (y!=0) (n=n-16 x=y) */
493  y = vec_sr(x, s);
494  nt = vec_sub(n,s);
495  m = (vui32_t)vec_cmpgt(y, z);
496  s = vec_sr(s,one);
497  x = vec_sel (x, y, m);
498  n = vec_sel (n, nt, m);
499 
500  /* y=x>>8 if (y!=0) (n=n-8 x=y) */
501  y = vec_sr(x, s);
502  nt = vec_sub(n,s);
503  m = (vui32_t)vec_cmpgt(y, z);
504  s = vec_sr(s,one);
505  x = vec_sel (x, y, m);
506  n = vec_sel (n, nt, m);
507 
508  /* y=x>>4 if (y!=0) (n=n-4 x=y) */
509  y = vec_sr(x, s);
510  nt = vec_sub(n,s);
511  m = (vui32_t)vec_cmpgt(y, z);
512  s = vec_sr(s,one);
513  x = vec_sel (x, y, m);
514  n = vec_sel (n, nt, m);
515 
516  /* y=x>>2 if (y!=0) (n=n-2 x=y) */
517  y = vec_sr(x, s);
518  nt = vec_sub(n,s);
519  m = (vui32_t)vec_cmpgt(y, z);
520  s = vec_sr(s,one);
521  x = vec_sel (x, y, m);
522  n = vec_sel (n, nt, m);
523 
524  /* y=x>>1 if (y!=0) return (n=n-2) */
525  y = vec_sr(x, s);
526  nt = vec_sub(n,s);
527  nt = vec_sub(nt,s);
528  m = (vui32_t)vec_cmpgt(y, z);
529  n = vec_sel (n, nt, m);
530 
531  /* else return (x-n) */
532  nt = vec_sub (n, x);
533  n = vec_sel (nt, n, m);
534  r = n;
535 #endif
536  return ((vui32_t) r);

763 static inline vi64_t
764 vec_mulesw (vi32_t a, vi32_t b)
765 {
766  vi64_t res;
767 #ifdef _ARCH_PWR8
768  // The vector vmulosw/vmulesw instructions introduced in PRW8
769 #if defined __GNUC__ && (__GNUC__ > 7)
770  res = vec_mule (a, b);
771 #else
772  __asm__(
773 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
774  "vmulosw %0,%1,%2;\n"
775 #else
776  "vmulesw %0,%1,%2;\n"
777 #endif
778  : "=v" (res)
779  : "v" (a),
780  "v" (b)
781  : );
782 #endif
783 #else
784  // must be PWR7 or older
785  vui32_t uia, uib;
786  vui32_t amask, bmask, t1, t2, r;
787  vui64_t ui_prod;
788  const vui32_t zero= { 0,0,0,0};
789 
790  uia = (vui32_t) a;
791  uib = (vui32_t) b;
792  // Generate 32-bit masks from the sign of each input word.
793  amask = (vui32_t) vec_srawi (a, 31);
794  bmask = (vui32_t) vec_srawi (b, 31);
795  // Extend the even masks to the right with zeros to form two 64-bit
796  // masks. We need the trailing zeros as the low 32-bits of the
797  // product are correct as-is and should not change.
798  amask = vec_mrgew (amask, zero);
799  bmask = vec_mrgew (bmask, zero);
800  // Compute the doubleword even unsigned word product
801  ui_prod = vec_muleuw (uia, uib);
802 
803  // Generate t1 = amask & b and t2 = bmask & a
804  t1 = vec_and (amask, uib);
805  t2 = vec_and (bmask, uia);
806  // Apply the correction res = ui_prod - t1 - t2
807  r = vec_sub ((vui32_t) ui_prod, t1);
808  res = (vi64_t) vec_sub (r, t2);
809 #endif
810  return (res);
811 }

1185 {

1187  vui32_t result;
1188 
1189  if (shb < 32)
1190  {
1191  /* Load the shift const in a vector. The element shifts require
1192  a shift amount for each element. For the immediate form the
1193  shift constant is splatted to all elements of the
1194  shift control. */
1195  if (__builtin_constant_p (shb) && (shb < 16))
1196  lshift = (vui32_t) vec_splat_s32(shb);

1202  result = vec_vslw (vra, lshift);

1206  result = vec_xor ((vui32_t) vra, (vui32_t) vra);
